`timescale 1ns / 1ps


module Divider(
    input clk,
    input rst,
    output reg oneHz_clk
    );
    parameter MAX_COUNT = 200_000_000 - 1;
    reg [27:0] counter_100M;
    initial begin
        oneHz_clk = 1;
    end
    //assign an = 6'b1110;
    always @(posedge clk) begin
    if  (counter_100M == MAX_COUNT) begin
        counter_100M <= 0;
        oneHz_clk = ~ oneHz_clk;
        end
    else begin
        counter_100M <= counter_100M + 1'b1;
        end
    end
endmodule
